def drive_forward(maze, current_position, direction):
    x, y = current_position
    if direction == 'up':
        next_position = (x, y - 1)
    elif direction == 'down':
        next_position = (x, y + 1)
    elif direction == 'left':
        next_position = (x - 1, y)
    else:  # direction == 'right'
        next_position = (x + 1, y)
    
    if maze[next_position[1]][next_position[0]] == 1:  # check if the next position is open
        return next_position
    else:
        return current_position  # if it's a wall, stay in the current position

def turn_left(direction):
    if direction == 'up':
        return 'left'
    elif direction == 'down':
        return 'right'
    elif direction == 'left':
        return 'down'
    else:  # direction == 'right'
        return 'up'
